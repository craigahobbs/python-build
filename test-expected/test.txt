if [ "$(docker images -q python:3.X)" = "" ]; then docker pull -q python:3.X; fi
docker run -i --rm -u `id -g`:`id -g` -v `pwd`:`pwd` -w `pwd` python:3.X python3 -m venv build/venv/test-python-3-X
docker run -i --rm -u `id -g`:`id -g` -v `pwd`:`pwd` -w `pwd` python:3.X build/venv/test-python-3-X/bin/pip -q --no-cache-dir install --progress-bar off --retries 2 --timeout 5 -U pip setuptools
docker run -i --rm -u `id -g`:`id -g` -v `pwd`:`pwd` -w `pwd` python:3.X build/venv/test-python-3-X/bin/pip -q --no-cache-dir install --progress-bar off --retries 2 --timeout 5 -e .
touch build/venv/test-python-3-X.build
